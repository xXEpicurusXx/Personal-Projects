    def save(self, *args, **kwargs):
        
        if self.id:
            existing = get_object_or_404(ImageDeleteTest, id = self.id)
            if existing.icon != self.icon:
                existing.icon.delete(save = False)
        
        super(ImageDeleteTest, self).save(*args, **kwargs)
